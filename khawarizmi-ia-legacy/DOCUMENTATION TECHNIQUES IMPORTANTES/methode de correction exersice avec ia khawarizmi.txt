// photo-quality-check.js — Vérification AVANT envoi IA (économise les coûts)

import { Image } from 'react-native';

class PhotoQualityCheck {

    static async verifier(imageUri) {

        // Vérification locale (pas d'API = gratuit)
        const diagnostics = [];

        // 1. Résolution minimum (1280×720 pour lire l'écriture)
        const { width, height } = await Image.getSize(imageUri);
        if (width < 1280 || height < 720) {
            diagnostics.push({
                probleme: 'RESOLUTION_FAIBLE',
                message:  '📸 La photo est trop petite. Rapproche ton téléphone.',
            });
        }

        // 2. Taille du fichier (si trop compressé = flou)
        const fileSize = await getFileSize(imageUri);
        if (fileSize < 200_000) {  // < 200KB = probablement flou
            diagnostics.push({
                probleme: 'COMPRESSION',
                message:  '📸 La photo semble floue. Tiens ton téléphone stable.',
            });
        }

        // 3. Luminosité (analyse basique des pixels)
        const brightness = await analyserLuminosite(imageUri);
        if (brightness < 0.3) {
            diagnostics.push({
                probleme: 'TROP_SOMBRE',
                message:  '💡 La photo est trop sombre. Active le flash ou approche-toi de la lumière.',
            });
        }

        return {
            qualite_ok:  diagnostics.length === 0,
            diagnostics: diagnostics,
        };
    }
}

// UTILISATION DANS L'APP :
const check = await PhotoQualityCheck.verifier(photoUri);

if (!check.qualite_ok) {
    // NE PAS envoyer à l'IA (économie)
    // Afficher le message à l'élève
    showAlert(check.diagnostics[0].message);
    return;
}

// Photo OK → envoyer à l'IA Vision
const evaluation = await evaluerCopieManuscrite(photoBase64, ...);
