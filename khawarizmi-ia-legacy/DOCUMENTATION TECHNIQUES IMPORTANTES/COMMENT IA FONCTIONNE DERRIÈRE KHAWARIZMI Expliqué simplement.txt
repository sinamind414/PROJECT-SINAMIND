// CE QUI SE PASSE DANS TON SERVEUR — Étape 3

async function evaluerReponse(userId, questionId, reponseEleve) {

    // 3A. Récupérer le contexte depuis la BASE DE DONNÉES (pas d'IA)
    const question = await db.query(
        `SELECT * FROM annales WHERE id = $1`, [questionId]
    );

    const profil = await db.query(
        `SELECT * FROM mastery_micro_concepts 
         WHERE user_id = $1 AND micro_concept_id = $2`,
        [userId, question.micro_concept_id]
    );

    const errorDna = await db.query(
        `SELECT * FROM error_dna 
         WHERE user_id = $1 AND micro_concept_id = $2
         ORDER BY count DESC LIMIT 3`,
        [userId, question.micro_concept_id]
    );

    // 3B. CONSTRUIRE LE PROMPT (les instructions pour l'IA)
    // ══════════════════════════════════════════════════════
    // C'est ICI que ta "recette secrète" est.
    // L'IA (DeepSeek ou GPT) reçoit ces instructions.
    // Elle ne "pense" pas par elle-même.
    // Elle SUIT tes instructions.

    const prompt = `
Tu es KHAWARIZMI, correcteur expert du BAC algérien en Mathématiques.

EXERCICE : ${question.question_text}

SOLUTION OFFICIELLE :
${JSON.stringify(question.solution_officielle)}

RÉPONSE DE L'ÉLÈVE :
"${reponseEleve}"

PROFIL DE L'ÉLÈVE :
- Niveau actuel sur ce concept : ${profil.niveau}/4
- Erreurs fréquentes : ${errorDna.map(e => e.description_exemple).join('; ')}

ÉVALUE selon ces critères :
1. Le raisonnement est-il correct ? (70% de la note)
2. Le résultat est-il correct ? (30% de la note)
3. Les étapes sont-elles dans le bon ordre ?
4. Y a-t-il des erreurs de calcul ?

RÉPONDS EN JSON :
{
    "score": <0 à 100>,
    "etapes_correctes": [<liste des étapes bien faites>],
    "etapes_incorrectes": [<liste des erreurs avec explication>],
    "type_erreur_dominant": "<TYPE_1 ou TYPE_2 ou TYPE_3 ou TYPE_4 ou AUCUNE>",
    "feedback_court": "<1 phrase pour l'élève>",
    "feedback_detaille": "<explication de ce qui est bien et ce qui manque>"
}

RÈGLES :
→ Accepte les variantes mathématiques équivalentes
→ Ne pénalise PAS les fautes d'orthographe
→ Sois bienveillant mais précis
→ Mentionne spécifiquement ce qui est BIEN (pas juste les erreurs)
    `;

    // 3C. APPEL À L'IA (c'est ici que DeepSeek / OpenAI travaille)
    // ═══════════════════════════════════════════════════════════════

    const reponseIA = await fetch('https://api.deepseek.com/v1/chat/completions', {
        method: 'POST',
        headers: {
            'Content-Type':  'application/json',
            'Authorization': `Bearer ${process.env.DEEPSEEK_API_KEY}`,
        },
        body: JSON.stringify({
            model:    'deepseek-chat',
            messages: [{ role: 'user', content: prompt }],
            temperature: 0.1,  // Bas = réponse cohérente, pas créative
            response_format: { type: 'json_object' },
        }),
    });

    const evaluation = await reponseIA.json();
    const resultat   = JSON.parse(evaluation.choices[0].message.content);

    // 3D. METTRE À JOUR LA BASE DE DONNÉES (pas d'IA)
    // ═══════════════════════════════════════════════════

    // Sauvegarder le résultat
    await db.query(`
        INSERT INTO exercise_results
            (user_id, question_id, micro_concept_id,
             student_answer, score, ai_feedback, error_type)
        VALUES ($1, $2, $3, $4, $5, $6, $7)
    `, [
        userId,
        questionId,
        question.micro_concept_id,
        reponseEleve,
        resultat.score / 100,
        JSON.stringify(resultat),
        resultat.type_erreur_dominant,
    ]);

    // Mettre à jour SM-2 (ALGORITHME, pas d'IA)
    const sm2Result = SM2.calculate(
        SM2.scoreToQuality(resultat.score),
        profil.easiness_factor,
        profil.interval_jours,
        profil.repetitions
    );

    await db.query(`
        UPDATE mastery_micro_concepts
        SET score_actuel       = $1,
            easiness_factor    = $2,
            interval_jours     = $3,
            repetitions        = $4,
            prochaine_revision = $5,
            updated_at         = NOW()
        WHERE user_id = $6 AND micro_concept_id = $7
    `, [
        resultat.score / 100,
        sm2Result.ef,
        sm2Result.interval,
        sm2Result.reps,
        sm2Result.nextDate,
        userId,
        question.micro_concept_id,
    ]);

    // Mettre à jour l'ADN d'erreur (pas d'IA)
    if (resultat.type_erreur_dominant !== 'AUCUNE') {
        await ErrorDNA.updateProfile(userId, {
            micro_concept_id:  question.micro_concept_id,
            error_type:        resultat.type_erreur_dominant,
            description:       resultat.etapes_incorrectes[0],
        });
    }

    // Attribuer XP (pas d'IA)
    await Gamification.awardXP(userId,
        resultat.score >= 60 ? 'drill_correct' : 'drill_wrong'
    );

    // 3E. RENVOYER LE RÉSULTAT AU TÉLÉPHONE
    return {
        score:            resultat.score,
        feedback:         resultat.feedback_court,
        details:          resultat.feedback_detaille,
        etapes_correctes: resultat.etapes_correctes,
        erreurs:          resultat.etapes_incorrectes,
        xp_gagne:         resultat.score >= 60 ? 25 : 5,
        prochaine_revision: sm2Result.nextDate,
    };
}

// error-autopsy.js — PAS D'IA. Juste du code.

function diagnostiquerErreur(q1, q2, q3) {

    if (q1 === false) {
        return {
            type:      'TYPE_1',
            nom:       'Erreur de concept',
            action:    'FEYNMAN',
            message:   'Tu ne maîtrises pas encore ce concept. Explique-le-moi avec tes mots avant de refaire l\'exercice.',
            sm2_reset: true,
        };
    }

    if (q1 === true && q2 === false) {
        return {
            type:      'TYPE_2',
            nom:       'Erreur de méthode',
            action:    'DRILL_METHODE',
            message:   'Tu connais le concept mais tu ne sais pas quand l\'utiliser. Je te prépare des exercices "Quelle méthode ici ?"',
            sm2_reset: false,
        };
    }

    if (q1 === true && q2 === true && q3 === true) {
        return {
            type:      'TYPE_3',
            nom:       'Erreur d\'exécution',
            action:    'REVOIR_CALCUL',
            message:   'Tu sais faire ! C\'est juste un piège de calcul. Regarde où tu t\'es trompé.',
            sm2_reset: false,
        };
    }

    return {
        type:      'TYPE_4',
        nom:       'Erreur de panique',
        action:    'EXAM_SIMULE',
        message:   'Tu savais faire mais tu as bloqué. C\'est l\'anxiété d\'examen. On va s\'entraîner en conditions réelles.',
        sm2_reset: false,
    };
}

async function evaluerFeynman(concept, explicationEleve, matiere) {

    const prompt = `
Tu es KHAWARIZMI, expert pédagogique du BAC algérien en ${matiere}.

L'élève devait expliquer "${concept}" comme à un enfant de 10 ans.

SON EXPLICATION :
"${explicationEleve}"

ÉVALUE sur 10 :
1. Simplicité (3 pts) : A-t-il évité le jargon ?
2. Justesse (4 pts) : L'idée est-elle correcte ?
3. Exemple (2 pts) : A-t-il donné un exemple concret ?
4. Complétude (1 pt) : Les points essentiels sont-ils là ?

RÉPONDS EN JSON :
{
    "score": <0-10>,
    "comprend_vraiment": <true si score >= 7>,
    "ce_qui_est_bien": "<ce que l'élève a bien compris>",
    "ce_qui_manque": "<ce qui manque ou est imprécis>",
    "question_suivante": "<une question pour approfondir>"
}
    `;

    const reponse = await fetch('https://api.openai.com/v1/chat/completions', {
        method: 'POST',
        headers: {
            'Content-Type':  'application/json',
            'Authorization': `Bearer ${process.env.OPENAI_API_KEY}`,
        },
        body: JSON.stringify({
            model:           'gpt-4o-mini',
            messages:        [{ role: 'user', content: prompt }],
            temperature:     0.2,
            response_format: { type: 'json_object' },
        }),
    });

    return await reponse.json();
}
