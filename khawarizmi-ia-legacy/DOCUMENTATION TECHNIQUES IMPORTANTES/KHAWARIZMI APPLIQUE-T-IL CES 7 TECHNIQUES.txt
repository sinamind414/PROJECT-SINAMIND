                       mmc.retrievability
                FROM annales a
                JOIN mastery_micro_concepts mmc
                    ON mmc.micro_concept_id = a.micro_concept_id
                WHERE mmc.user_id = $1
                  AND a.chapitre_id = $2
                  AND mmc.prochaine_revision <= NOW()
                ORDER BY mmc.retrievability ASC
                LIMIT 3
            """, [userId, chapitre_id])

            questions.extend(dues.rows)

        # Mélanger aléatoirement (pas groupé par chapitre)
        import random
        random.shuffle(questions)

        return {
            'mode':       'INTERLEAVING',
            'questions':  questions,
            'message':    """

async def evaluer_schema_photo(imageBase64, schema_id, openai):
    """
    L'IA compare la photo du schéma de l'élève
    avec le schéma officiel du programme algérien.
    """
    schema_ref = SCHEMAS_SVT_PROGRAMME_ALGERIEN.get(schema_id)

    if not schema_ref:
        return {'erreur': 'Schéma non trouvé dans la base'}

    prompt = f"""
Tu es KHAWARIZMI, expert du BAC algérien en Sciences Naturelles.

L'élève devait reproduire ce schéma de mémoire :
{schema_ref['schema_ascii']}

Points critiques à vérifier absolument :
{chr(10).join(f'→ {p}' for p in schema_ref['points_critiques'])}

Photo du schéma de l'élève ci-jointe.

ÉVALUE :
1. Les flèches sont-elles dans le bon sens ?
2. Les termes scientifiques sont-ils exacts ?
   (Plasmocyte ≠ "cellule qui fabrique anticorps")
3. L'ordre des étapes est-il correct ?
4. Y a-t-il des éléments manquants ?

FORMAT JSON :
{{
    "score": <0-10>,
    "fleches_correctes": true/false,
    "vocabulaire_exact": true/false,
    "ordre_correct": true/false,
    "elements_manquants": ["<element 1>", "<element 2>"],
    "feedback": "<message bienveillant et précis>",
    "question_socratique": "<une question pour approfondir>"
}}

RÈGLE : Commence par ce qui est BIEN dans le schéma.
"""

    response = await openai.chat.completions.create(
        model='gpt-4o-mini',
        messages=[{
            'role': 'user',
            'content': [
                {'type': 'text', 'text': prompt},
                {'type': 'image_url', 'image_url': {
                    'url': f'data:image/jpeg;base64,{imageBase64}',
                    'detail': 'high',
                }},
            ],
        }],
        temperature=0.1,
        response_format={'type': 'json_object'},
    )

    return response.choices[0].message.content
